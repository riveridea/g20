/*
Implement pow(x, n).
*/

public class Solution {
    public double pow(double x, int n) {
        if(n == 0) return 1;
        if(x == 0) return 0;
        if(n>0) return 1.0/pow(x,-n);
        else{
            double s=pow(x,n/2);
            if(n%2 != 0) return s*s/x;
            else return s*s;
        }       
    }
}
